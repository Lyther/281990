country_event = {
	id = tt_reform_plane_ai.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		is_ai = yes
		has_ascension_perk = ap_tt_internal_revolution
		NOT = { has_country_flag = flag_tt_internal_revolution_finished }
	}
	immediate = {
		country_event = { id = tt_reform_plane_ai.1 days = 30 }
		if = {
			limit = {
				NOT = { has_country_flag = flag_tt_internal_revolution }
				resource_stockpile_compare = { resource = unity value >= 5000 }
			}
			add_resource = { unity = -5000 }
			add_modifier = { modifier = "mod_tt_internal_revolution_1" days = -1 }
			set_country_flag = flag_tt_internal_revolution
		}
		else_if = {
			limit = {
				has_modifier = mod_tt_internal_revolution_1
				resource_stockpile_compare = { resource = unity value >= 10000 }
			}
			add_resource = { unity = -10000 }
			remove_modifier = "mod_tt_internal_revolution_1"
			add_modifier = { modifier = "mod_tt_internal_revolution_2" days = -1 }
		}
		else_if = {
