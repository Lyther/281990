	set_variable = { which = tradition_loafing_2_effect_var_$SHIP_MODIFIER$ value = 0 }

	if = {
		limit = { check_variable = { which = tradition_loafing_2_effect_var_$SHIP_MODIFIER$ value > -1 } }

		add_modifier = { modifier = $TRADITION_MODIFIER$ multiplier = tradition_loafing_2_effect_var_$SHIP_MODIFIER$ days = -1 }
	}
	clear_variable = tradition_loafing_2_effect_var_$SHIP_MODIFIER$
}
