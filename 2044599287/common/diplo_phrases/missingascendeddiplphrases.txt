		}

				}
			}
		}
		03_INSULT_ACCEPT_Ascendancy = {
			trigger = {
				is_gestalt = no
				OR = {
					is_country_type = ascended_empire
					is_country_type = lost_empire
				}
			}
		}
		01_INSULT_ACCEPT_Ascendancy_Overlord = {
			weight = {
				weight = 2500
			}

				is_gestalt = no
				OR = {
					is_country_type = ascended_empire
					is_country_type = lost_empire
