		multiply = 2
		# 负相界旅者：x2
		has_policy_flag = horror_dimension_traveler
	}
	modifier = {
		multiply = 2
		# 超越因果+负相奥秘：2倍
		has_country_flag = dimension_traveler_end
		has_ascension_perk = ap_horror
	}

		multiply = 1.5
		# 超越因果+负相奥秘+时空因果：3倍
		has_country_flag = dimension_traveler_end
		has_ascension_perk = ap_time
	}
	modifier = {
		multiply = 2
		# 始终因果：2倍
		NOT = { has_country_flag = dimension_traveler_end }
	}
}
