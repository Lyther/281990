	id = parallel_sky.11
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		exists = event_target:parallel_sky_country #未击败平行帝国
		is_same_value = event_target:parallel_sky_planet
		OR = {
			colonizable_planet = no #星球被毁灭
			NOT = { has_deposit = d_parallel_sky } #无特殊地貌
		}
	}
	
	immediate = {
		if = {
			limit = {
				colonizable_planet = no
			}
			remove_deposit = d_parallel_sky
			event_target:parallel_sky_country = {
				destroy_country = yes
			}
			random_country = {
				limit = {
					has_origin = origin_parallel_sky
				}
				country_event = { id = parallel_sky.12 }
			}
		}
		else = {
			add_deposit = d_parallel_sky
		}
	}
}
