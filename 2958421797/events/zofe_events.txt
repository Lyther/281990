#Hedonist Decision Very Bad Outcome
planet_event = {
	id = zofe.3004
	title = zofe.3004.name
	desc = zofe.3004.desc
	picture = GFX_evt_fallen_empire
	
	is_triggered_only = yes
	
	option = {
		name = zofe.3004.a
		if = {
			limit = {
				exists = owner
				owner = { NOT = { has_authority = auth_corporate } }
			}
			custom_tooltip = decadence_demotion
		}
		if = {
			limit = {
				exists = owner
				owner = { has_authority = auth_corporate }
			}
			custom_tooltip = decadence_demotion_megacorp
		}
		hidden_effect = {
			if = {
				limit = {
					exists = owner
					owner = {
						NOT = { has_authority = auth_corporate }
					}
					any_owned_pop = { has_job = politician }
				}
				random_owned_pop = {
					limit = { has_job = politician }
					unemploy_pop = yes
					clear_pop_category = yes #Drops them to worker-class
				}
			}
			
			if = {
				limit = {
					exists = owner
					owner = {
						has_authority = auth_corporate
					}
					any_owned_pop = { has_job = executive }
				}
				random_owned_pop = {
					limit = { has_job = executive }
					unemploy_pop = yes
					clear_pop_category = yes #Drops them to worker-class
				}
			}
			
			every_owned_pop = {
				limit = { has_pop_flag = decadent_pop }
				kill_pop = yes
			}
		}
		##Happiness Modifier
		add_modifier = {
			modifier = decadence_unhappiness_2
			days = 720
		}
	}
}
