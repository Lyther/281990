namespace = fallen_weaponry

event = {
	id = fallen_weaponry.1
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
	
		every_country = {
			limit = { 
				is_country_type = fallen_empire
				has_ethic = ethic_fanatic_spiritualist
			}
			every_owned_planet = {
				limit = {
					NOT = { has_planet_flag = fallen_spiritualist_capital }
					is_capital = yes
				}
				set_planet_flag = fallen_spiritualist_capital
			}
			if = {
				limit = {
					NOT = { exists = event_target:fallen_spiritualist }
				}
				save_global_event_target_as = fallen_spiritualist
			}
		}
		
		every_country = {
			limit = { 
				is_country_type = fallen_empire
				has_ethic = ethic_fanatic_materialist
			}
			every_owned_planet = {
				limit = {
					NOT = { has_planet_flag = fallen_materialist_capital }
					is_capital = yes
				}
				set_planet_flag = fallen_materialist_capital
			}
			random_owned_planet = {
				limit = {
					NOT = { has_planet_flag = fallen_materialist_archives }
					has_building = building_master_archive
					is_capital = no
				}
				set_planet_flag = fallen_materialist_archives
			}
			if = {
				limit = {
					NOT = { exists = event_target:fallen_materialist }
				}
				save_global_event_target_as = fallen_materialist
			}
		}
		
		every_country = {
			limit = { 
				is_country_type = fallen_empire
				has_ethic = ethic_fanatic_xenophile
			}
			every_owned_planet = {
				limit = {
					NOT = { has_planet_flag = fallen_xenophile_capital }
					is_capital = yes
				}
				set_planet_flag = fallen_xenophile_capital
			}
			if = {
				limit = {
					NOT = { exists = event_target:fallen_xenophile }
				}
				save_global_event_target_as = fallen_xenophile
			}
		}
		
		every_country = {
			limit = { 
				is_country_type = fallen_empire
				has_ethic = ethic_fanatic_xenophobe
			}
			every_owned_planet = {
				limit = {
					NOT = { has_planet_flag = fallen_xenophobe_capital }
					is_capital = yes
				}
				set_planet_flag = fallen_xenophobe_capital
			}
			if = {
				limit = {
					NOT = { exists = event_target:fallen_xenophobe }
				}
				save_global_event_target_as = fallen_xenophobe
			}
		}
		
		every_country = {
			limit = { 
				is_country_type = fallen_empire

country_event = {
	id = fallen_weaponry.100
	title = fallen_weaponry.100.name
	desc = fallen_weaponry.100.desc
	picture = GFX_evt_underground_city
	location = from
	show_sound = event_mystic_reveal

	fire_only_once = yes

	trigger = {
		host_has_dlc = "Ancient Relics Story Pack"
		masters_of_warfare = yes		
#		is_ai = no # For Testing
		is_ascended_empire = yes
		has_technology = tech_fallen_technologies_4
		has_technology = tech_fallen_technologies_6

		any_owned_planet = { has_planet_flag = fallen_spiritualist_capital }
	
		NOT = { has_country_flag = fallen_spiritualist_capital_raided }
	}
	
	mean_time_to_happen = {
		months = 5
	}
	
	immediate = {
		set_country_flag = fallen_spiritualist_capital_raided
		
		random_owned_planet = {
			limit = {
#				is_capital = yes # For Testing
				has_planet_flag = fallen_spiritualist_capital
			}
			save_event_target_as = fallen_spiritualist_digsite
		}
	}
	
	option = {
		name = fallen_weaponry.100.a
		event_target:fallen_spiritualist_digsite = { 
			create_archaeological_site = fallen_spiritualist_digsite_1
		}
		owner = {
			if = {
				limit = { 
					NOT = { has_event_chain = lost_technologies_chain }
				}
				begin_event_chain = {
					event_chain = "lost_technologies_chain"
					target = this
				}
			}
		}
		owner = {
			add_monthly_resource_mult = {
				resource = engineering_research
     			value = @tier1researchreward
     			min = @tier1researchmin
     			max = @tier1researchmax
     		}
		}
	}
}
