	}

			NOT = { is_country_type = ascended_empire }
		}
		set_country_type = ascended_empire	
	}
	
	#ZoFE Country Type Change - Lost Empire
	option = {
		name = "zofe_debug.103.lost_empire"
		trigger = {
			NOT = { is_country_type = lost_empire }
		}
		set_country_type = lost_empire
	}
	
	#ZoFE Country Type Change - Default
	option = {
		name = "zofe_debug.103.default"	
		trigger = {
			NOT = { is_country_type = default }
		}
		set_country_type = default
	}

	#Adds 10 Decadent Pops
	option = {
		name = "zofe_debug.103.hedonist_10"	
		trigger = {
			OR = {
				is_country_type = ascended_empire
				is_country_type = lost_empire
			}
		}
		custom_tooltip = decadent_pop_add_10
		if = {
			limit = { is_gestalt = no }
			while = {
				count = 10
				random_owned_pop = {
					limit = {
						is_sapient = yes
						has_citizenship_rights = yes
						is_pop_category = worker
						is_enslaved = no
						is_being_purged = no
						is_robot_pop = no
						is_being_assimilated = no
						NOT = { has_pop_flag = decadent_pop }
						NOT = { has_job = ae_hedonist }
					}
					set_pop_flag = decadent_pop
				}
			}
		}
		if = {
			limit = { has_authority = auth_hive_mind }
			while = {
				count = 10
				random_owned_pop = {
					limit = {
						is_pop_category = simple_drone
						NOT = { has_pop_flag = decadent_pop }
						NOT = { has_job = ae_confused_drone }
					}
					set_pop_flag = decadent_pop
				}
			}
		}
		if = {
			limit = { has_authority = auth_machine_intelligence }
			while = {
				count = 10
				random_owned_pop = {
					limit = {
						is_pop_category = simple_drone
						NOT = { has_pop_flag = decadent_pop }
						NOT = { has_job = ae_maintenance_bot }
					}
					set_pop_flag = decadent_pop
				}
			}
		}
	}
	
	option = {
		name = "zofe_menu.101.exit"
		hidden_effect = {
			country_event = { id = zofe_debug.1 days = 0 }
		}
	}
}
