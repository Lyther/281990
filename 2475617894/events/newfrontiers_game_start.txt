#Awakened Start
event = {
	id = newfrontiers.17
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		any_playable_country = { has_origin = origin_awakened }
	}


	immediate = {
		random_country = {
			limit = { has_origin = origin_awakened }
			capital_scope = {	
				while = {
					limit = { num_pops > 1 }
					random_owned_pop = { kill_pop = yes }
				}
			}
			
			every_galaxy_planet = {
				limit = {
					OR = {
						has_planet_flag = awakened_second_planet
						has_planet_flag = awakened_third_planet
					}
				}
				set_owner = prev
					
				create_pop = {
					species = prev.owner_main_species
				}
					
				solar_system = {
					if = {
						limit = { exists = starbase }
						starbase = {
							set_owner = prev.owner
						}
					}
					else = {
						create_starbase = {
							size = starbase_starport
							module = shipyard
							owner = prev.owner
						}
					}
				}
			}
			
			while = {
				count = 10
				give_random_technology_hidden = yes
			}
			
			every_owned_planet = {
				remove_all_districts = yes
				remove_all_buildings = yes
				clear_deposits = yes
				
				# 5 + 3 energy
				add_deposit = d_searing_desert
				add_deposit = d_underwater_vent
				add_deposit = d_tempestous_mountain
				if = {
					limit = { owner = { is_machine_empire = yes } }
					add_deposit = d_underwater_vent
				}

				# 5 + 3 mining
				add_deposit = d_ore_rich_caverns
				add_deposit = d_prosperous_mesa
				add_deposit = d_submerged_ore_veins

				# 5 + 3 farming (lithoids swap 3 for minerals)
				add_deposit = d_lush_jungle
				add_deposit = d_teeming_reef
				
				if = {
					limit = { owner = { is_machine_empire = no } }
					if = {
						limit = { owner = { is_lithoid_empire = yes } }
						add_deposit = d_rich_mountain
					}
					else = {
						add_deposit = d_black_soil
					}
				}
				
				add_deposit = d_relic_landfill
				add_deposit = d_ancient_ruins
				add_deposit = d_sarcophagus
				
				if = {
					limit = { owner = { is_regular_empire = yes } }
					add_building = building_capital
				}
				if = {
					limit = { owner = { is_hive_empire = yes } }
					add_building = building_hive_capital
